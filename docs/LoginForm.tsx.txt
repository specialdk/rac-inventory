LoginForm.tsx

import React, { useState } from 'react';
import { useEffect } from 'react';
import { supabase } from '../../lib/supabase';
import { useAuth } from '../../context/AuthContext';
import { toast } from 'react-hot-toast';
import { LogIn, Key } from 'lucide-react';
import FormField from '../ui/FormField';
import PasswordResetForm from './PasswordResetForm';

interface LoginFormProps {
  onToggleForm: () => void;
}

const LoginForm: React.FC<LoginFormProps> = ({ onToggleForm }) => {
  const { signIn } = useAuth();
  const [email, setEmail] = useState('');
  const [password, setPassword] = useState('');
  const [isLoading, setIsLoading] = useState(false);
  const [showPasswordReset, setShowPasswordReset] = useState(false);
  const [showNewPasswordForm, setShowNewPasswordForm] = useState(false);
  const [newPassword, setNewPassword] = useState('');
  const [confirmPassword, setConfirmPassword] = useState('');
  const [showManualReset, setShowManualReset] = useState(false);
  const [resetEmail, setResetEmail] = useState('');
  const [resetToken, setResetToken] = useState('');
  const [showAdminBypass, setShowAdminBypass] = useState(false);
  const [adminEmail, setAdminEmail] = useState('');
  const [adminNewPassword, setAdminNewPassword] = useState('');
  const [adminConfirmPassword, setAdminConfirmPassword] = useState('');

  useEffect(() => {
    // Check if this is a password reset callback
    const checkForPasswordReset = () => {
      console.log('=== PASSWORD RESET DEBUG ===');
      console.log('Full URL:', window.location.href);
      console.log('Hash:', window.location.hash);
      console.log('Pathname:', window.location.pathname);
      
      // Check if this came from a reset route or has reset parameters
      if (window.location.pathname === '/reset' || window.location.hash.includes('type=recovery')) {
        console.log('Password reset route detected');
        // Check for valid token in URL params or hash
        const urlParams = new URLSearchParams(window.location.search);
        const hashParams = new URLSearchParams(window.location.hash.substring(1));
        
        const accessToken = urlParams.get('access_token') || hashParams.get('access_token');
        const type = urlParams.get('type') || hashParams.get('type');
        
        if (type === 'recovery' && accessToken) {
          setShowNewPasswordForm(true);
          return;
        }
      }
      
      // Parse hash parameters - handle malformed URLs with multiple # symbols
      let hashString = window.location.hash.substring(1); // Remove first #
      
      if (!hashString) return;
      
      // Fix malformed URLs with multiple # symbols by replacing # with &
      hashString = hashString.replace(/#/g, '&');
      
      console.log('Cleaned hash string:', hashString);
      
      const hashParams = new URLSearchParams(hashString);
      
      const accessToken = hashParams.get('access_token');
      const type = hashParams.get('type');
      const error = hashParams.get('error');
      const errorDescription = hashParams.get('error_description');
      const errorCode = hashParams.get('error_code');
      
      console.log('Parsed reset params:', { 
        type, 
        accessToken: accessToken ? 'present' : 'missing', 
        error, 
        errorDescription,
        errorCode,
        allParams: Object.fromEntries(hashParams)
      });
      
      // Check for errors
      if (error) {
        console.log('Password reset error detected:', error);
        if (error === 'access_denied' || errorCode === 'otp_expired' || errorDescription?.includes('expired')) {
          toast.error('Password reset link has expired. Please request a new one.');
          setShowPasswordReset(true); // Automatically show the reset form
        } else {
          toast.error('Password reset failed. Please try again.');
        }
        // Clear the URL
        window.history.replaceState({}, document.title, window.location.pathname);
        return;
      }
      
      if (type === 'recovery' && accessToken) {
        // Valid reset token
        console.log('Valid reset token found, showing password form');
        toast.success('Please set your new password');
        setShowNewPasswordForm(true);
      } else {
        console.log('No valid reset token found', { type, hasToken: !!accessToken });
      }
    };
    
    checkForPasswordReset();
  }, []);

  const handlePasswordUpdate = async (e: React.FormEvent) => {
    e.preventDefault();
    
    if (newPassword !== confirmPassword) {
      toast.error('Passwords do not match');
      return;
    }
    
    if (newPassword.length < 6) {
      toast.error('Password must be at least 6 characters');
      return;
    }
    
    setIsLoading(true);
    
    try {
      const { error } = await supabase.auth.updateUser({
        password: newPassword
      });
      
      if (error) throw error;
      
      toast.success('Password updated successfully! You can now sign in.');
      setShowNewPasswordForm(false);
      
      // Clear the URL hash
      window.history.replaceState(null, '', window.location.pathname);
    } catch (error) {
      console.error('Password update error:', error);
      toast.error('Failed to update password');
    } finally {
      setIsLoading(false);
    }
  };

  const handleManualReset = async (e: React.FormEvent) => {
    e.preventDefault();
    
    if (!resetToken.trim()) {
      toast.error('Please enter the token from your email');
      return;
    }
    
    if (newPassword !== confirmPassword) {
      toast.error('Passwords do not match');
      return;
    }
    
    if (newPassword.length < 6) {
      toast.error('Password must be at least 6 characters');
      return;
    }
    
    setIsLoading(true);
    
    try {
      // Verify the token and update password
      const { error } = await supabase.auth.verifyOtp({
        email: resetEmail,
        token: resetToken,
        type: 'recovery'
      });
      
      if (error) throw error;
      
      // Update the password
      const { error: updateError } = await supabase.auth.updateUser({
        password: newPassword
      });
      
      if (updateError) throw updateError;
      
      toast.success('Password updated successfully! You can now sign in.');
      setShowManualReset(false);
      setShowNewPasswordForm(false);
      
    } catch (error: any) {
      console.error('Manual reset error:', error);
      if (error.message?.includes('expired')) {
        toast.error('Reset token has expired. Please request a new password reset.');
      } else {
        toast.error('Invalid token or reset failed. Please check the token and try again.');
      }
    } finally {
      setIsLoading(false);
    }
  };

  const handleAdminBypass = async (e: React.FormEvent) => {
    e.preventDefault();
    
    if (adminNewPassword !== adminConfirmPassword) {
      toast.error('Passwords do not match');
      return;
    }
    
    if (adminNewPassword.length < 6) {
      toast.error('Password must be at least 6 characters');
      return;
    }
    
    setIsLoading(true);
    
    try {
      // Create new admin account
      const { data, error } = await supabase.auth.signUp({
        email: adminEmail,
        password: adminNewPassword,
        options: {
          data: {
            full_name: 'Admin User'
          }
        }
      });
      
      if (error) {
        if (error.message.includes('already registered')) {
          // Try to sign in with existing account
          const { error: signInError } = await supabase.auth.signInWithPassword({
            email: adminEmail,
            password: adminNewPassword
          });
          
          if (signInError) {
            toast.error('Account exists but password is incorrect. Try the correct password or use password reset.');
            return;
          }
          
          toast.success('Successfully signed in with existing account!');
          setShowAdminBypass(false);
          return;
        } else {
          throw error;
        }
      }
      
      if (data.user) {
        // Check if email confirmation is required
        if (!data.session) {
          toast.success('Account created! Please check your email for verification link, or try signing in directly.');
          setShowAdminBypass(false);
          return;
        }
        
        // Account created and signed in successfully
        toast.success('Admin account created and signed in successfully!');
        setShowAdminBypass(false);
      }
      
    } catch (error: any) {
      console.error('Admin bypass error:', error);
      toast.error('Failed to create admin account: ' + error.message);
    } finally {
      setIsLoading(false);
    }
  };

  if (showPasswordReset) {
    return <PasswordResetForm onBack={() => setShowPasswordReset(false)} />;
  }
  
  if (showManualReset) {
    return (
      <div className="min-h-screen flex items-center justify-center bg-gray-50 py-12 px-4 sm:px-6 lg:px-8">
        <div className="max-w-md w-full space-y-8">
          <div>
            <h2 className="mt-6 text-center text-3xl font-extrabold text-gray-900">
              Manual Password Reset
            </h2>
            <p className="mt-2 text-center text-sm text-gray-600">
              Copy the token from your reset email and enter it below
            </p>
          </div>
          <form className="mt-8 space-y-6" onSubmit={handleManualReset}>
            <div className="space-y-4">
              <FormField
                name="resetEmail"
                label="Your Email"
                type="email"
                value={resetEmail}
                onChange={(e) => setResetEmail(e.target.value)}
                required
                placeholder="Enter your email address"
              />
              <div>
                <label className="block text-sm font-medium text-gray-700 mb-1">
                  Reset Token <span className="text-red-500">*</span>
                </label>
                <textarea
                  value={resetToken}
                  onChange={(e) => setResetToken(e.target.value)}
                  className="w-full px-3 py-2 border border-gray-300 rounded-lg focus:ring-2 focus:ring-orange-500 focus:border-transparent"
                  rows={3}
                  placeholder="Copy the long token from your reset email URL (the part after 'token=')"
                  required
                />
                <p className="mt-1 text-xs text-gray-500">
                  Look for "token=" in your email link and copy everything after it until the next "&"
                </p>
              </div>
              <FormField
                name="newPassword"
                label="New Password"
                type="password"
                value={newPassword}
                onChange={(e) => setNewPassword(e.target.value)}
                required
                placeholder="Enter new password"
              />
              <FormField
                name="confirmPassword"
                label="Confirm New Password"
                type="password"
                value={confirmPassword}
                onChange={(e) => setConfirmPassword(e.target.value)}
                required
                placeholder="Confirm new password"
              />
            </div>

            <div className="space-y-4">
              <button
                type="submit"
                disabled={isLoading}
                className="group relative w-full flex justify-center py-2 px-4 border border-transparent text-sm font-medium rounded-md text-white bg-orange-600 hover:bg-orange-700 focus:outline-none focus:ring-2 focus:ring-offset-2 focus:ring-orange-500 disabled:bg-orange-400"
              >
                {isLoading ? 'Updating...' : 'Update Password'}
              </button>
              <button
                type="button"
                onClick={() => setShowManualReset(false)}
                className="w-full text-center text-sm text-gray-600 hover:text-gray-900"
              >
                Back to Login
              </button>
            </div>
          </form>
        </div>
      </div>
    );
  }
  
  if (showAdminBypass) {
    return (
      <div className="min-h-screen flex items-center justify-center bg-gray-50 py-12 px-4 sm:px-6 lg:px-8">
        <div className="max-w-md w-full space-y-8">
          <div>
            <h2 className="mt-6 text-center text-3xl font-extrabold text-gray-900">
              Admin Bypass
            </h2>
            <p className="mt-2 text-center text-sm text-gray-600">
              Create a new admin account to access the system
            </p>
          </div>
          <form className="mt-8 space-y-6" onSubmit={handleAdminBypass}>
            <div className="space-y-4">
              <FormField
                name="adminEmail"
                label="New Admin Email"
                type="email"
                value={adminEmail}
                onChange={(e) => setAdminEmail(e.target.value)}
                required
                placeholder="Enter a new admin email"
              />
              <FormField
                name="adminNewPassword"
                label="New Password"
                type="password"
                value={adminNewPassword}
                onChange={(e) => setAdminNewPassword(e.target.value)}
                required
                placeholder="Enter new password"
              />
              <FormField
                name="adminConfirmPassword"
                label="Confirm New Password"
                type="password"
                value={adminConfirmPassword}
                onChange={(e) => setAdminConfirmPassword(e.target.value)}
                required
                placeholder="Confirm new password"
              />
            </div>

            <div className="space-y-4">
              <button
                type="submit"
                disabled={isLoading}
                className="group relative w-full flex justify-center py-2 px-4 border border-transparent text-sm font-medium rounded-md text-white bg-red-600 hover:bg-red-700 focus:outline-none focus:ring-2 focus:ring-offset-2 focus:ring-red-500 disabled:bg-red-400"
              >
                {isLoading ? 'Creating...' : 'Create Admin Account'}
              </button>
              <button
                type="button"
                onClick={() => setShowAdminBypass(false)}
                className="w-full text-center text-sm text-gray-600 hover:text-gray-900"
              >
                Back to Login
              </button>
            </div>
          </form>
        </div>
      </div>
    );
  }
  
  if (showNewPasswordForm) {
    return (
      <div className="min-h-screen flex items-center justify-center bg-gray-50 py-12 px-4 sm:px-6 lg:px-8">
        <div className="max-w-md w-full space-y-8">
          <div>
            <h2 className="mt-6 text-center text-3xl font-extrabold text-gray-900">
              Set New Password
            </h2>
            <p className="mt-2 text-center text-sm text-gray-600">
              Enter your new password below
            </p>
          </div>
          <form className="mt-8 space-y-6" onSubmit={handlePasswordUpdate}>
            <div className="space-y-4">
              <FormField
                name="newPassword"
                label="New Password"
                type="password"
                value={newPassword}
                onChange={(e) => setNewPassword(e.target.value)}
                required
                placeholder="Enter new password"
              />
              <FormField
                name="confirmPassword"
                label="Confirm New Password"
                type="password"
                value={confirmPassword}
                onChange={(e) => setConfirmPassword(e.target.value)}
                required
                placeholder="Confirm new password"
              />
            </div>

            <div>
              <button
                type="submit"
                disabled={isLoading}
                className="group relative w-full flex justify-center py-2 px-4 border border-transparent text-sm font-medium rounded-md text-white bg-orange-600 hover:bg-orange-700 focus:outline-none focus:ring-2 focus:ring-offset-2 focus:ring-orange-500 disabled:bg-orange-400"
              >
                {isLoading ? 'Updating...' : 'Update Password'}
              </button>
            </div>
          </form>
        </div>
      </div>
    );
  }

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    setIsLoading(true);

    try {
      await signIn(email, password);
      toast.success('Welcome back!');
    } catch (error) {
      console.error('Login error:', error);
      const errorMessage = (error as any)?.message || 'Invalid email or password';
      
      if (errorMessage.includes('Invalid login credentials')) {
        toast.error('No account found with these credentials. Use Admin Bypass to create your first account.');
      } else if (errorMessage.includes('Email not confirmed')) {
        toast.error('Please check your email and click the verification link before signing in.');
      } else {
        toast.error(errorMessage);
      }
    } finally {
      setIsLoading(false);
    }
  };

  return (
    <div className="min-h-screen flex items-center justify-center bg-gray-50 py-12 px-4 sm:px-6 lg:px-8">
      <div className="max-w-md w-full space-y-8">
        <div>
          <h2 className="mt-6 text-center text-3xl font-extrabold text-gray-900">
            Sign in to your account
          </h2>
          <p className="mt-2 text-center text-sm text-gray-600">
            Rirratjingu Mining Inventory System
          </p>
          <div className="mt-4 p-3 bg-blue-50 border border-blue-200 rounded-md">
            <p className="text-sm text-blue-800 text-center">
              <strong>First time setup?</strong> Use "Admin Bypass" below to create your first account.
            </p>
          </div>
        </div>
        <form className="mt-8 space-y-6" onSubmit={handleSubmit}>
          <div className="rounded-md shadow-sm space-y-4">
            <FormField
              name="email"
              label="Email address"
              type="email"
              value={email}
              onChange={(e) => setEmail(e.target.value)}
              required
              autoComplete="email"
            />
            <FormField
              name="password"
              label="Password"
              type="password"
              value={password}
              onChange={(e) => setPassword(e.target.value)}
              required
              autoComplete="current-password"
            />
          </div>

          <div>
            <button
              type="submit"
              disabled={isLoading}
              className="group relative w-full flex justify-center py-2 px-4 border border-transparent text-sm font-medium rounded-md text-white bg-orange-600 hover:bg-orange-700 focus:outline-none focus:ring-2 focus:ring-offset-2 focus:ring-orange-500 disabled:bg-orange-400"
            >
              <LogIn className="h-5 w-5 mr-2" />
              {isLoading ? 'Signing in...' : 'Sign in'}
            </button>
          </div>
          
          <div className="space-y-2">
            <button
              type="button"
              onClick={() => setShowPasswordReset(true)}
              className="w-full text-center text-sm text-orange-600 hover:text-orange-900"
            >
              Forgot your password?
            </button>
            <button
              type="button"
              onClick={() => setShowManualReset(true)}
              className="w-full text-center text-sm text-blue-600 hover:text-blue-900 flex items-center justify-center gap-1"
            >
              <Key className="h-4 w-4" />
              Manual Password Reset
            </button>
            <button
              type="button"
              onClick={() => setShowAdminBypass(true)}
              className="w-full text-center text-sm text-red-600 hover:text-red-900 flex items-center justify-center gap-1"
            >
              🚨 Admin Bypass (Emergency)
            </button>
            <button
              type="button"
              onClick={onToggleForm}
              className="w-full text-center text-sm text-gray-600 hover:text-gray-900"
            >
              Need an account? Sign up
            </button>
          </div>
        </form>
      </div>
    </div>
  );
};

export default LoginForm
