ProductForm.tsx

import React, { useState, useEffect } from 'react';
import { Save } from 'lucide-react';
import { toast } from 'react-hot-toast';
import FormField from '../../ui/FormField';
import { Product } from '../../../lib/types';
import { saveProduct } from '../../../lib/api';

interface ProductFormProps {
  onClose: () => void;
  onSave: () => void;
  editingItem?: Product | null;
}

const ProductForm: React.FC<ProductFormProps> = ({ onClose, onSave, editingItem }) => {
  const [formData, setFormData] = useState<Partial<Product>>({
    product_code: '',
    product_name: '',
    family_group: '',
    unit: 'tonnes',
    standard_cost: 0,
    min_stock_level: 0,
    max_stock_level: 0,
    active: true
  });
  
  const [errors, setErrors] = useState<Record<string, string>>({});
  const [isSubmitting, setIsSubmitting] = useState(false);

  useEffect(() => {
    if (editingItem) {
      setFormData(editingItem);
    } else {
      setFormData({
        product_code: '',
        product_name: '',
        family_group: '',
        unit: 'tonnes',
        standard_cost: 0,
        min_stock_level: 0,
        max_stock_level: 0,
        active: true
      });
    }
  }, [editingItem]);

  const handleChange = (e: React.ChangeEvent<HTMLInputElement | HTMLSelectElement | HTMLTextAreaElement>) => {
    const { name, value, type } = e.target;
    
    // Convert numeric inputs to numbers
    if (type === 'number') {
      setFormData({ ...formData, [name]: parseFloat(value) || 0 });
    } else {
      setFormData({ ...formData, [name]: value });
    }
    
    // Clear error when field is changed
    if (errors[name]) {
      setErrors({ ...errors, [name]: '' });
    }
  };

  const validateForm = (): boolean => {
    const newErrors: Record<string, string> = {};
    
    // Required fields
    if (!formData.product_code) newErrors.product_code = 'Product code is required';
    if (!formData.product_name) newErrors.product_name = 'Product name is required';
    if (!formData.family_group) newErrors.family_group = 'Family group is required';
    if (!formData.unit) newErrors.unit = 'Unit is required';
    
    // Numeric validation
    if (formData.standard_cost === undefined || formData.standard_cost < 0) {
      newErrors.standard_cost = 'Standard cost must be a positive number';
    }
    if (formData.min_stock_level === undefined || formData.min_stock_level < 0) {
      newErrors.min_stock_level = 'Minimum stock level must be a positive number';
    }
    if (formData.max_stock_level === undefined || formData.max_stock_level < 0) {
      newErrors.max_stock_level = 'Maximum stock level must be a positive number';
    }
    
    // Validate min/max relationship
    if (formData.min_stock_level && formData.max_stock_level && formData.min_stock_level > formData.max_stock_level) {
      newErrors.min_stock_level = 'Minimum stock level cannot be greater than maximum stock level';
    }
    
    setErrors(newErrors);
    return Object.keys(newErrors).length === 0;
  };

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    
    if (!validateForm()) return;
    
    setIsSubmitting(true);
    
    try {
      await saveProduct(formData);
      toast.success(editingItem ? 'Product updated successfully' : 'Product added successfully');
      onSave();
    } catch (error) {
      console.error('Error saving product:', error);
      toast.error('Failed to save product');
    } finally {
      setIsSubmitting(false);
    }
  };

  return (
    <form onSubmit={handleSubmit} className="space-y-4">
      <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
        <FormField
          name="product_code"
          label="Product Code"
          type="text"
          value={formData.product_code || ''}
          onChange={handleChange}
          required
          placeholder="e.g., A20-STD"
          error={errors.product_code}
        />
        
        <FormField
          name="product_name"
          label="Product Name"
          type="text"
          value={formData.product_name || ''}
          onChange={handleChange}
          required
          error={errors.product_name}
        />
        
        <FormField
          name="family_group"
          label="Family Group"
          type="select"
          value={formData.family_group || ''}
          onChange={handleChange}
          required
          options={['AGGREGATES', 'SAND', 'ROAD_BASE', 'DUST', 'PREMIX', 'FILL']}
          error={errors.family_group}
        />
        
        <FormField
          name="unit"
          label="Unit"
          type="select"
          value={formData.unit || ''}
          onChange={handleChange}
          required
          options={['tonnes', 'cubic metres']}
          error={errors.unit}
        />
        
        <FormField
          name="standard_cost"
          label="Standard Cost ($)"
          type="number"
          value={formData.standard_cost || ''}
          onChange={handleChange}
          required
          min={0}
          step="0.01"
          error={errors.standard_cost}
        />
        
        <FormField
          name="min_stock_level"
          label="Minimum Stock Level"
          type="number"
          value={formData.min_stock_level || ''}
          onChange={handleChange}
          required
          min={0}
          step="0.01"
          error={errors.min_stock_level}
        />
        
        <FormField
          name="max_stock_level"
          label="Maximum Stock Level"
          type="number"
          value={formData.max_stock_level || ''}
          onChange={handleChange}
          required
          min={0}
          step="0.01"
          error={errors.max_stock_level}
        />
      </div>
      
      <div className="flex justify-end space-x-3 mt-6">
        <button
          type="button"
          onClick={onClose}
          className="px-4 py-2 text-gray-700 bg-gray-200 rounded-lg hover:bg-gray-300 transition-colors"
          disabled={isSubmitting}
        >
          Cancel
        </button>
        <button
          type="submit"
          className="px-4 py-2 bg-orange-600 text-white rounded-lg hover:bg-orange-700 transition-colors flex items-center gap-2 disabled:bg-orange-400 disabled:cursor-not-allowed"
          disabled={isSubmitting}
        >
          <Save className="h-4 w-4" />
          {isSubmitting ? 'Saving...' : 'Save Product'}
        </button>
      </div>
    </form>
  );
};

export default ProductForm;
