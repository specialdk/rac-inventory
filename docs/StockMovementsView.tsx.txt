StockMovementsView.tsx

import React, { useState } from 'react';
import { Plus, Search, FileSpreadsheet, FileDown } from 'lucide-react';
import { useAppContext } from '../../context/AppContext';
import { utils, writeFile } from 'xlsx';
import { format } from 'date-fns';

interface StockMovementsViewProps {
  onAddMovement: () => void;
}

const StockMovementsView: React.FC<StockMovementsViewProps> = ({ onAddMovement }) => {
  const { stockMovements, products, getProductName, getLocationName, getCustomerName } = useAppContext();
  const [searchTerm, setSearchTerm] = useState('');

  const handleExport = (type: 'xlsx' | 'csv') => {
    // Prepare data for export
    const data = stockMovements.map(movement => ({
      'Date': movement.movement_date,
      'Type': movement.movement_type,
      'Product Code': products.find(p => p.id === movement.product_id)?.product_code || '',
      'Product Name': getProductName(movement.product_id),
      'Location': getLocationName(movement.location_id),
      'Quantity': movement.quantity,
      'Total Value': movement.total_value,
      'Reference': movement.reference_number,
      'Docket': movement.docket_number || '',
      'Customer': movement.customer_id ? getCustomerName(movement.customer_id) : ''
    }));

    // Create workbook
    const wb = utils.book_new();
    const ws = utils.json_to_sheet(data);

    // Add worksheet to workbook
    utils.book_append_sheet(wb, ws, 'Stock Movements');

    // Generate filename with current date
    const dateStr = format(new Date(), 'yyyy-MM-dd');
    const filename = `stock-movements-${dateStr}.${type}`;

    // Export file
    writeFile(wb, filename);
  };

  const filteredMovements = stockMovements.filter(movement => 
    !searchTerm || 
    getProductName(movement.product_id).toLowerCase().includes(searchTerm.toLowerCase()) ||
    (movement.reference_number?.toLowerCase().includes(searchTerm.toLowerCase()) ?? false)
  );

  return (
    <div className="space-y-6">
      <div className="flex justify-between items-center">
        <h2 className="text-2xl font-bold text-gray-900">Stock Movements</h2>
        <div className="flex gap-4">
          <button
            onClick={() => handleExport('xlsx')}
            className="flex items-center gap-2 px-4 py-2 text-sm font-medium text-green-700 bg-green-50 rounded-lg hover:bg-green-100"
          >
            <FileSpreadsheet className="h-4 w-4" />
            Export Excel
          </button>
          <button
            onClick={() => handleExport('csv')}
            className="flex items-center gap-2 px-4 py-2 text-sm font-medium text-blue-700 bg-blue-50 rounded-lg hover:bg-blue-100"
          >
            <FileDown className="h-4 w-4" />
            Export CSV
          </button>
          <div className="relative">
            <Search className="absolute left-3 top-1/2 transform -translate-y-1/2 text-gray-400 h-4 w-4" />
            <input
              type="text"
              placeholder="Search movements..."
              className="pl-10 pr-4 py-2 border border-gray-300 rounded-lg focus:ring-2 focus:ring-orange-500 focus:border-transparent"
              value={searchTerm}
              onChange={(e) => setSearchTerm(e.target.value)}
            />
          </div>
          <button
            onClick={onAddMovement}
            className="bg-orange-600 text-white px-6 py-2 rounded-lg hover:bg-orange-700 transition-colors flex items-center gap-2"
          >
            <Plus className="h-4 w-4" />
            Add Movement
          </button>
        </div>
      </div>

      <div className="bg-white rounded-lg shadow-sm overflow-hidden">
        <div className="overflow-y-auto max-h-[calc(100vh-300px)] overflow-x-hidden">
          <table className="min-w-full divide-y divide-gray-200">
            <thead className="bg-gray-50">
              <tr>
                <th className="sticky top-0 bg-gray-50 px-3 py-2 text-left text-xs font-medium text-gray-500 uppercase w-24">Date</th>
                <th className="sticky top-0 bg-gray-50 px-3 py-2 text-left text-xs font-medium text-gray-500 uppercase w-24">Type</th>
                <th className="sticky top-0 bg-gray-50 px-3 py-2 text-left text-xs font-medium text-gray-500 uppercase">Product</th>
                <th className="sticky top-0 bg-gray-50 px-3 py-2 text-left text-xs font-medium text-gray-500 uppercase">Location</th>
                <th className="sticky top-0 bg-gray-50 px-3 py-2 text-left text-xs font-medium text-gray-500 uppercase w-32">Quantity</th>
                <th className="sticky top-0 bg-gray-50 px-3 py-2 text-left text-xs font-medium text-gray-500 uppercase w-24">Unit $</th>
                <th className="sticky top-0 bg-gray-50 px-3 py-2 text-left text-xs font-medium text-gray-500 uppercase w-28">Total $</th>
                <th className="sticky top-0 bg-gray-50 px-3 py-2 text-left text-xs font-medium text-gray-500 uppercase">Reference</th>
              </tr>
            </thead>
            <tbody className="bg-white divide-y divide-gray-200">
              {filteredMovements.map((movement) => (
                <tr key={movement.id} className="hover:bg-gray-50">
                  <td className="px-3 py-2 whitespace-nowrap text-sm text-gray-900">
                    {movement.movement_date}
                  </td>
                  <td className="px-3 py-2 whitespace-nowrap">
                    <span className={`inline-flex px-2 py-1 text-xs font-semibold rounded-full ${
                      movement.movement_type === 'OPENING' ? 'bg-blue-100 text-blue-800' :
                      movement.movement_type === 'PRODUCTION' ? 'bg-green-100 text-green-800' :
                      movement.movement_type === 'SALES' ? 'bg-red-100 text-red-800' :
                      movement.movement_type === 'TRANSFER' ? 'bg-yellow-100 text-yellow-800' :
                      'bg-gray-100 text-gray-800'
                    }`}>
                      {movement.movement_type}
                    </span>
                  </td>
                  <td className="px-3 py-2 text-sm text-gray-900 truncate max-w-[200px]">
                    {getProductName(movement.product_id)}
                  </td>
                  <td className="px-3 py-2 text-sm text-gray-900 truncate max-w-[200px]">
                    {getLocationName(movement.location_id)}
                  </td>
                  <td className="px-3 py-2 whitespace-nowrap">
                    <span className={`text-sm font-medium ${
                      movement.quantity < 0 ? 'text-red-600' : 'text-green-600'
                    }`}>
                      {movement.quantity > 0 ? '+' : ''}{movement.quantity.toFixed(3)}
                    </span>
                  </td>
                  <td className="px-3 py-2 whitespace-nowrap text-sm text-gray-900">
                    ${movement.unit_cost.toFixed(2)}
                  </td>
                  <td className="px-3 py-2 whitespace-nowrap">
                    <span className={`text-sm font-medium ${
                      movement.total_value < 0 ? 'text-red-600' : 'text-green-600'
                    }`}>
                      ${movement.total_value.toLocaleString()}
                    </span>
                  </td>
                  <td className="px-3 py-2 text-sm text-gray-900">
                    {movement.reference_number}
                    {movement.docket_number && (
                      <div className="text-xs text-gray-500">Docket: {movement.docket_number}</div>
                    )}
                    {movement.customer_id && (
                      <div className="text-xs text-gray-500">{getCustomerName(movement.customer_id)}</div>
                    )}
                  </td>
                </tr>
              ))}
            </tbody>
          </table>
        </div>
      </div>
    </div>
  );
};

export default StockMovementsView;
