OperationsView.tsx

import React, { useState, useEffect } from 'react';
import { Package, TrendingUp, DollarSign, Plus, Save, X, Search, Calendar, Truck, User, MapPin, AlertCircle, CheckCircle } from 'lucide-react';
import { useAppContext } from '../../context/AppContext';
import { saveStockMovement } from '../../lib/api';
import { format } from 'date-fns';
import FormField from '../ui/FormField';
import Modal from '../ui/Modal';
import { toast } from 'react-hot-toast';

const OperationsView: React.FC = () => {
  const { 
    products, 
    locations, 
    customers, 
    vehicles, 
    drivers,
    stockMovements,
    currentStock,
    refreshStockMovements,
    refreshCurrentStock,
    getProductName,
    getLocationName,
    getCustomerName,
    getDriverName
  } = useAppContext();

  // State for operation type
  const [activeOperation, setActiveOperation] = useState('Production');
  const [showForm, setShowForm] = useState(false);
  const [formData, setFormData] = useState<any>({});
  const [searchTerm, setSearchTerm] = useState('');
  const [lastUsedOperator, setLastUsedOperator] = useState<number | null>(null);
  const [showWarningModal, setShowWarningModal] = useState(false);
  const [pendingLocationChange, setPendingLocationChange] = useState<number | null>(null);
  const [lastOperator, setLastOperator] = useState<string>(
    localStorage.getItem('lastOperator') || ''
  );

  // Form configurations
  const productionFormFields = [
    { name: 'movement_date', label: 'Production Date', type: 'date', required: true, defaultValue: format(new Date(), 'yyyy-MM-dd') },
    { name: 'product_id', label: 'Product', type: 'select', options: products.map(p => ({ value: p.id, label: `${p.product_code} - ${p.product_name}` })), required: true },
    { name: 'from_location_id', label: 'From (Production Area)', type: 'select', options: locations.filter(l => l.location_type === 'PRODUCTION').map(l => ({ value: l.id, label: l.location_name })), required: true },
    { name: 'to_location_id', label: 'To (Stockpile)', type: 'select', options: locations.filter(l => l.location_type === 'STOCKPILE').map(l => ({ value: l.id, label: l.location_name })), required: true },
    { name: 'quantity', label: 'Quantity (tonnes)', type: 'number', step: '0.001', required: true, placeholder: 'e.g., 25.500' },
  //  { name: 'unit_cost', label: 'Production Cost per Tonne ($)', type: 'number', step: '0.01', required: true, placeholder: 'e.g., 35.00' },
    { name: 'unit_cost', label: 'Production Cost per Tonne ($)', type: 'number', step: '0.01', required: true, placeholder: 'Auto-filled from product price', disabled: true }, 
    { name: 'vehicle_id', label: 'Loader/Vehicle', type: 'select', options: vehicles.map(v => ({ value: v.id, label: `${v.vehicle_code} (${v.registration})` })), required: false },
    { name: 'driver_id', label: 'Operator', type: 'select', options: drivers.map(d => ({ value: d.id, label: `${d.driver_code} - ${d.first_name} ${d.last_name}` })), required: false },
    { name: 'reference_number', label: 'Production Reference', type: 'text', required: true, placeholder: 'e.g., PROD-001' },
    { name: 'notes', label: 'Notes', type: 'textarea', required: false, placeholder: 'Production notes, weather conditions, etc.' }
  ];

  const salesFormFields = [
    { name: 'movement_date', label: 'Sale Date', type: 'date', required: true, defaultValue: format(new Date(), 'yyyy-MM-dd') },
    { name: 'product_id', label: 'Product Sold', type: 'select', options: products.map(p => ({ value: p.id, label: `${p.product_code} - ${p.product_name}` })), required: true },
    { name: 'from_location_id', label: 'From Stockpile', type: 'select', options: locations.filter(l => l.location_type === 'STOCKPILE').map(l => ({ value: l.id, label: l.location_name })), required: true },
    { name: 'quantity', label: 'Quantity Sold (tonnes)', type: 'number', step: '0.001', required: true, placeholder: 'e.g., 15.750' },
    { name: 'unit_cost', label: 'Sale Price per Tonne ($)', type: 'number', step: '0.01', required: true, placeholder: 'e.g., 45.00' },
    { name: 'customer_id', label: 'Customer', type: 'select', options: customers.map(c => ({ value: c.id, label: `${c.customer_code} - ${c.customer_name}` })), required: true },
    { name: 'vehicle_id', label: 'Delivery Vehicle', type: 'select', options: vehicles.map(v => ({ value: v.id, label: `${v.vehicle_code} (${v.registration})` })), required: false },
    { name: 'driver_id', label: 'Driver', type: 'select', options: drivers.map(d => ({ value: d.id, label: `${d.driver_code} - ${d.first_name} ${d.last_name}` })), required: false },
    { name: 'docket_number', label: 'Docket Number', type: 'text', required: true, placeholder: 'e.g., 60561' },
    { name: 'reference_number', label: 'Sale Reference', type: 'text', required: true, placeholder: 'e.g., SALE-001' },
    { name: 'notes', label: 'Notes', type: 'textarea', required: false, placeholder: 'Delivery instructions, special requirements, etc.' }
  ];

  // Form handlers
  const handleAddOperation = (type: string) => {
    setActiveOperation(type);
    setFormData({
      movement_date: format(new Date(), 'yyyy-MM-dd'),
      from_location_id: locations.find(l => l.location_type === 'PRODUCTION')?.id || '',
      vehicle_id: vehicles.find(v => v.vehicle_code === 'RAC-003')?.id || ''
    });
    setShowForm(true);
  };

  // Set last operator when form opens
  useEffect(() => {
    if (lastOperator) {
      setFormData(prev => ({...prev, driver_id: lastOperator}));
    }
  }, [showForm, lastOperator]);

  // Replace your current handleProductChange function with this improved version:
  const handleProductChange = (e: React.ChangeEvent<HTMLSelectElement>) => {
  const productId = parseInt(e.target.value);
  const product = products.find(p => p.id === productId);
  const assignedLocation = locations.find(l => 
    l.assigned_product_id === productId && 
    l.location_type === 'STOCKPILE' && 
    l.active
  );
  
  // Find RAC-001 vehicle or fall back to first available
  const defaultVehicle = vehicles.find(v => v.vehicle_code === 'RAC-001') || vehicles[0];
  
  // For production: set default production location (00-09 range)
  const defaultProductionLocation = locations.find(l => 
    l.location_type === 'PRODUCTION' && 
    l.location_code >= '00' &&
    l.location_code <= '09' &&
    l.active
  );
  
  setFormData({
    ...formData,
    product_id: productId,
    // Auto-select correct stockpile for production
    to_location_id: assignedLocation?.id || formData.to_location_id || '',
    // Set default production area for "from" location
    from_location_id: activeOperation === 'Production' 
      ? (defaultProductionLocation?.id || formData.from_location_id)
      : (assignedLocation?.id || ''), // For sales, use assigned stockpile as "from"
    // Default to RAC-001 vehicle
    vehicle_id: defaultVehicle?.id || '',
    // AUTO-POPULATE PRODUCTION COST FROM PRODUCT PRICE
    unit_cost: product?.current_price || 0,
    // Persist last operator (already handled in useEffect)
  });
};

  // Handle stockpile location change
  const handleStockpileChange = (e: React.ChangeEvent<HTMLSelectElement>) => {
    const newLocationId = parseInt(e.target.value);
    const newLocation = locations.find(l => l.id === newLocationId);
    const selectedProduct = products.find(p => p.id === parseInt(formData.product_id));
    
    if (newLocation?.assigned_product_id && 
        selectedProduct && 
        newLocation.assigned_product_id !== selectedProduct.id) {
      const existingProduct = products.find(p => p.id === newLocation.assigned_product_id);
      setPendingLocationChange(newLocationId);
      setShowWarningModal(true);
    } else {
      setFormData({ ...formData, to_location_id: newLocationId });
    }
  };

  // Handle stockpile override confirmation
  const handleConfirmLocationOverride = () => {
    if (pendingLocationChange) {
      setFormData({ ...formData, to_location_id: pendingLocationChange });
    }
    setShowWarningModal(false);
    setPendingLocationChange(null);
  };

  // Handle stockpile override cancellation
  const handleCancelLocationOverride = () => {
    const assignedLocation = locations.find(l => l.assigned_product_id === parseInt(formData.product_id));
    setFormData({ ...formData, to_location_id: assignedLocation?.id || '' });
    setShowWarningModal(false);
    setPendingLocationChange(null);
  };

  const handleSaveOperation = async () => {
    try {
      // Validate mixed products
      if (activeOperation === 'Production') {
        const locationId = parseInt(formData.to_location_id);
        const productId = parseInt(formData.product_id);
        
        const existingStock = currentStock.find(s => 
          s.location_id === locationId && 
          s.product_id !== productId && 
          s.current_quantity > 0
        );
        
        if (existingStock) {
          const existingProduct = products.find(p => p.id === existingStock.product_id);
          const selectedProduct = products.find(p => p.id === productId);
          
          toast.error(
            `Cannot mix products in stockpile!\n` +
            `Location contains ${existingProduct?.product_name || 'another product'}\n` +
            `Cannot add ${selectedProduct?.product_name || 'new product'}`
          );
          return;
        }
      }

      const movementData = {
        movement_date: formData.movement_date,
        movement_type: activeOperation.toUpperCase(),
        product_id: parseInt(formData.product_id),
        location_id: parseInt(activeOperation === 'Production' ? formData.to_location_id : formData.from_location_id),
        quantity: activeOperation === 'Production' 
          ? parseFloat(formData.quantity)  // Allow negative quantities for production
          : activeOperation === 'Sales' 
            ? -Math.abs(parseFloat(formData.quantity))  // Force negative for sales
            : parseFloat(formData.quantity),
        unit_cost: parseFloat(formData.unit_cost),
        reference_number: formData.reference_number,
        customer_id: formData.customer_id ? parseInt(formData.customer_id) : null,
        vehicle_id: formData.vehicle_id ? parseInt(formData.vehicle_id) : null,
        driver_id: formData.driver_id ? parseInt(formData.driver_id) : null,
        docket_number: formData.docket_number,
        notes: formData.notes,
        created_by: 'OPERATOR'
      };

      await saveStockMovement(movementData);
      await refreshCurrentStock();
      await refreshStockMovements();
      
      // Store last operator
      if (formData.driver_id) {
        setLastOperator(formData.driver_id.toString());
        localStorage.setItem('lastOperator', formData.driver_id.toString());
      }
      
      toast.success('Movement added successfully');
      setShowForm(false);
    } catch (error) {
      console.error('Error saving operation:', error);
      toast.error(`Failed to save ${activeOperation.toLowerCase()}`);
    }
  };

  // Get current form fields based on operation type
  const getCurrentFormFields = () => {
    return activeOperation === 'Production' ? productionFormFields : salesFormFields;
  };

  // Render operation buttons
  const renderOperationButtons = () => (
    <div className="grid grid-cols-1 md:grid-cols-2 gap-6 mb-8">
      {/* Production Button */}
      <div className="bg-gradient-to-r from-orange-500 to-orange-600 rounded-lg shadow-lg p-6 text-white">
        <div className="flex items-center justify-between">
          <div>
            <h3 className="text-xl font-bold mb-2">Production Entry</h3>
            <p className="text-orange-100 mb-4">Record daily production from extraction to stockpiles</p>
            <button
              onClick={() => handleAddOperation('Production')}
              className="bg-white text-orange-600 px-6 py-3 rounded-lg font-semibold hover:bg-orange-50 transition-colors flex items-center gap-2"
            >
              <TrendingUp className="h-5 w-5" />
              Add Production
            </button>
          </div>
          <Package className="h-16 w-16 text-orange-200" />
        </div>
      </div>

      {/* Sales Button */}
      <div className="bg-gradient-to-r from-blue-500 to-blue-600 rounded-lg shadow-lg p-6 text-white">
        <div className="flex items-center justify-between">
          <div>
            <h3 className="text-xl font-bold mb-2">Sales Entry</h3>
            <p className="text-blue-100 mb-4">Record customer sales and deliveries</p>
            <button
              onClick={() => handleAddOperation('Sales')}
              className="bg-white text-blue-600 px-6 py-3 rounded-lg font-semibold hover:bg-blue-50 transition-colors flex items-center gap-2"
            >
              <DollarSign className="h-5 w-5" />
              Add Sale
            </button>
          </div>
          <DollarSign className="h-16 w-16 text-blue-200" />
        </div>
      </div>
    </div>
  );

  // Calculate today's metrics
  const today = format(new Date(), 'yyyy-MM-dd');
  const todaysMovements = stockMovements.filter(m => m.movement_date === today);
  const todaysProduction = todaysMovements
    .filter(m => m.movement_type === 'PRODUCTION')
    .reduce((sum, m) => sum + m.quantity, 0);
  const todaysSales = todaysMovements
    .filter(m => m.movement_type === 'SALES')
    .reduce((sum, m) => sum + Math.abs(m.quantity), 0);
  const activeVehiclesCount = vehicles.filter(v => v.status === 'ACTIVE' && v.active).length;
  const activeDriversCount = drivers.filter(d => d.status === 'ACTIVE').length;

  return (
    <div className="space-y-6">
      <div className="flex justify-between items-center">
        <h2 className="text-2xl font-bold text-gray-900">Quarry Operations</h2>
        <p className="text-gray-600">Record daily production and sales activities</p>
      </div>

      {/* Operation Buttons */}
      {renderOperationButtons()}

      {/* Quick Stats */}
      <div className="grid grid-cols-1 md:grid-cols-4 gap-4 mb-8">
        <div className="bg-white rounded-lg shadow-sm p-4 border-l-4 border-orange-500">
          <div className="flex items-center">
            <TrendingUp className="h-8 w-8 text-orange-600" />
            <div className="ml-3">
              <p className="text-sm font-medium text-gray-600">Today's Production</p>
              <p className="text-xl font-bold text-gray-900">{todaysProduction.toFixed(1)} tonnes</p>
            </div>
          </div>
        </div>

        <div className="bg-white rounded-lg shadow-sm p-4 border-l-4 border-blue-500">
          <div className="flex items-center">
            <DollarSign className="h-8 w-8 text-blue-600" />
            <div className="ml-3">
              <p className="text-sm font-medium text-gray-600">Today's Sales</p>
              <p className="text-xl font-bold text-gray-900">{todaysSales.toFixed(1)} tonnes</p>
            </div>
          </div>
        </div>

        <div className="bg-white rounded-lg shadow-sm p-4 border-l-4 border-purple-500">
          <div className="flex items-center">
            <Truck className="h-8 w-8 text-purple-600" />
            <div className="ml-3">
              <p className="text-sm font-medium text-gray-600">Active Vehicles</p>
              <p className="text-xl font-bold text-gray-900">{activeVehiclesCount}</p>
            </div>
          </div>
        </div>

        <div className="bg-white rounded-lg shadow-sm p-4 border-l-4 border-orange-500">
          <div className="flex items-center">
            <User className="h-8 w-8 text-orange-600" />
            <div className="ml-3">
              <p className="text-sm font-medium text-gray-600">Active Operators</p>
              <p className="text-xl font-bold text-gray-900">{activeDriversCount}</p>
            </div>
          </div>
        </div>
      </div>

      {/* Recent Movements Table */}
      <div className="bg-white rounded-lg shadow-sm">
        <div className="px-6 py-4 border-b border-gray-200 flex justify-between items-center">
          <h2 className="text-xl font-semibold text-gray-900">Recent Movements</h2>
          <div className="relative">
            <Search className="absolute left-3 top-1/2 transform -translate-y-1/2 text-gray-400 h-4 w-4" />
            <input
              type="text"
              placeholder="Search movements..."
              className="pl-10 pr-4 py-2 border border-gray-300 rounded-lg focus:ring-2 focus:ring-orange-500 focus:border-transparent"
              value={searchTerm}
              onChange={(e) => setSearchTerm(e.target.value)}
            />
          </div>
        </div>
        <div className="overflow-x-auto">
          <table className="min-w-full divide-y divide-gray-200">
            <thead className="bg-gray-50">
              <tr>
                <th className="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase">Date</th>
                <th className="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase">Type</th>
                <th className="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase">Product</th>
                <th className="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase">Quantity</th>
                <th className="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase">Location</th>
                <th className="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase">Customer/Operator</th>
                <th className="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase">Reference</th>
                <th className="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase">Status</th>
              </tr>
            </thead>
            <tbody className="bg-white divide-y divide-gray-200">
              {stockMovements
                .filter(movement => 
                  !searchTerm || 
                  getProductName(movement.product_id).toLowerCase().includes(searchTerm.toLowerCase()) ||
                  movement.reference_number?.toLowerCase().includes(searchTerm.toLowerCase()) ||
                  (movement.customer_id && getCustomerName(movement.customer_id).toLowerCase().includes(searchTerm.toLowerCase()))
                )
                .map((movement) => (
                <tr key={movement.id} className="hover:bg-gray-50">
                  <td className="px-6 py-4 whitespace-nowrap text-sm text-gray-900">
                    {movement.movement_date}
                  </td>
                  <td className="px-6 py-4 whitespace-nowrap">
                    <span className={`inline-flex px-2 py-1 text-xs font-semibold rounded-full ${
                      movement.movement_type === 'PRODUCTION' ? 'bg-green-100 text-green-800' : 'bg-blue-100 text-blue-800'
                    }`}>
                      {movement.movement_type}
                    </span>
                  </td>
                  <td className="px-6 py-4 whitespace-nowrap text-sm text-gray-900">
                    {getProductName(movement.product_id)}
                  </td>
                  <td className="px-6 py-4 whitespace-nowrap">
                    <span className={`text-sm font-medium ${
                      movement.quantity > 0 ? 'text-green-600' : 'text-red-600'
                    }`}>
                      {movement.quantity > 0 ? '+' : ''}{movement.quantity.toFixed(3)}
                    </span>
                  </td>
                  <td className="px-6 py-4 whitespace-nowrap text-sm text-gray-900">
                    {getLocationName(movement.location_id)}
                  </td>
                  <td className="px-6 py-4 whitespace-nowrap text-sm text-gray-900">
                    {movement.customer_id ? getCustomerName(movement.customer_id) : 
                     movement.driver_id ? getDriverName(movement.driver_id) : '-'}
                  </td>
                  <td className="px-6 py-4 whitespace-nowrap text-sm text-gray-900">
                    {movement.reference_number}
                    {movement.docket_number && (
                      <div className="text-xs text-gray-500">Docket: {movement.docket_number}</div>
                    )}
                  </td>
                  <td className="px-6 py-4 whitespace-nowrap">
                    <span className="inline-flex items-center px-2 py-1 text-xs font-semibold rounded-full bg-green-100 text-green-800">
                      <CheckCircle className="h-3 w-3 mr-1" />
                      Completed
                    </span>
                  </td>
                </tr>
              ))}
            </tbody>
          </table>
        </div>
      </div>

      {/* Operation Form Modal */}
      <Modal
        title={`${activeOperation} Entry`}
        isOpen={showForm}
        onClose={() => setShowForm(false)}
        size="lg"
      >
        <div className="space-y-4">
          {getCurrentFormFields().map((field) => (
            <FormField
              key={field.name}
              name={field.name}
              label={field.label}
              type={field.type}
              value={formData[field.name] || field.defaultValue || ''}
              onChange={(e) => {
                if (field.name === 'product_id') {
                  handleProductChange(e);
                } else {
                  setFormData({ ...formData, [field.name]: e.target.value });
                }
              }}
              required={field.required}
              options={field.options}
              step={field.step}
              placeholder={field.placeholder}
            />
          ))}

          <div className="flex justify-end space-x-3 mt-6">
            <button
              onClick={() => setShowForm(false)}
              className="px-4 py-2 text-gray-700 bg-gray-200 rounded-lg hover:bg-gray-300 transition-colors"
            >
              Cancel
            </button>
            <button
              onClick={handleSaveOperation}
              className={`px-4 py-2 text-white rounded-lg transition-colors flex items-center gap-2 ${
                activeOperation === 'Production' 
                  ? 'bg-orange-600 hover:bg-orange-700' 
                  : 'bg-blue-600 hover:bg-blue-700'
              }`}
            >
              <Save className="h-4 w-4" />
              Save {activeOperation}
            </button>
          </div>
        </div>
      </Modal>
      
      {/* Warning Modal */}
      <Modal
        title="Warning: Stockpile Assignment Mismatch"
        isOpen={showWarningModal}
        onClose={handleCancelLocationOverride}
        size="md"
      >
        <div className="space-y-4">
          <div className="flex items-start space-x-3 text-yellow-700 bg-yellow-50 p-4 rounded-lg">
            <AlertCircle className="h-5 w-5 mt-0.5 flex-shrink-0" />
            <div>
              <p className="font-medium">Stockpile Assignment Warning</p>
              <p className="mt-1 text-sm">
                This stockpile is assigned to a different product. 
                Are you sure you want to place material in this location?
              </p>
            </div>
          </div>
          
          <div className="flex justify-end space-x-3">
            <button
              onClick={handleCancelLocationOverride}
              className="px-4 py-2 text-gray-700 bg-gray-200 rounded-lg hover:bg-gray-300 transition-colors"
            >
              Cancel
            </button>
            <button
              onClick={handleConfirmLocationOverride}
              className="px-4 py-2 bg-yellow-600 text-white rounded-lg hover:bg-yellow-700 transition-colors"
            >
              Override Assignment
            </button>
          </div>
        </div>
      </Modal>
    </div>
  );
};

export default OperationsView;
