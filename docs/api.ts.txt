import { supabase } from './supabase';
import { 
  Product, Location, Customer, Vehicle, Driver, 
  CurrentStock, StockMovement
} from './types';

// Products
export const getProducts = async (): Promise<Product[]> => {
  const { data, error } = await supabase
    .from('products')
    .select('*')
    .order('family_group', { ascending: true })
    .order('product_name', { ascending: true });
  
  if (error) throw error;
  return data || [];
};

export const saveProduct = async (product: Partial<Product>): Promise<Product> => {
  if (product.id) {
    // Update existing product
    const { data, error } = await supabase
      .from('products')
      .update(product)
      .eq('id', product.id)
      .select()
      .single();
    
    if (error) throw error;
    return data;
  } else {
    // Create new product
    const { data, error } = await supabase
      .from('products')
      .insert(product)
      .select()
      .single();
    
    if (error) throw error;
    return data;
  }
};

export const deleteProduct = async (id: number): Promise<void> => {
  const { error } = await supabase
    .from('products')
    .delete()
    .eq('id', id);
  
  if (error) throw error;
};

// Current Stock
export const getCurrentStock = async (): Promise<CurrentStock[]> => {
  const { data, error } = await supabase
    .from('current_stock')
    .select(`
      *,
      product:product_id(id, product_code, product_name, family_group, unit, min_stock_level, max_stock_level),
      location:location_id(id, location_code, location_name, location_type)
    `)
    .order('product_id', { ascending: true })

  return data || [];
};

// Stock Movements
export const getStockMovements = async (): Promise<StockMovement[]> => {
  const { data, error } = await supabase
    .from('stock_movements')
    .select(`
      *,
      product:product_id(id, product_code, product_name, family_group),
      location:location_id(id, location_code, location_name),
      customer:customer_id(id, customer_code, customer_name),
      vehicle:vehicle_id(id, vehicle_code, registration),
      driver:driver_id(id, driver_code, first_name, last_name)
    `)
    .order('movement_date', { ascending: false })
    .order('id', { ascending: false })
    .limit(100);
  
  if (error) throw error;
  return data || [];
};

export const saveStockMovement = async (movement: Partial<StockMovement>): Promise<StockMovement> => {
  // Remove total_value from the payload since it's a generated column
  const { total_value, ...movementData } = movement;
  
  if (movement.id) {
    // Update existing movement
    const { data, error } = await supabase
      .from('stock_movements')
      .update(movementData)
      .eq('id', movement.id)
      .select()
      .single();
    
    if (error) throw error;
    return data;
  } else {
    // Create new movement
    const { data, error } = await supabase
      .from('stock_movements')
      .insert(movementData)
      .select()
      .single();
    
    if (error) throw error;
    return data;
  }
};

// Dashboard queries
export const getDashboardMetrics = async () => {
  // Get count of products with stock
  const { data: productsWithStock, error: productsError } = await supabase
    .from('current_stock')
    .select('product_id')
    .gt('current_quantity', 0)
    .limit(1000);
  
  if (productsError) throw productsError;
  
  // Get total inventory value
  const { data: inventoryValue, error: inventoryError } = await supabase
    .from('current_stock')
    .select('current_quantity, average_cost');
  
  if (inventoryError) throw inventoryError;
  
  const totalValue = (inventoryValue || []).reduce(
    (sum, item) => sum + (item.current_quantity * item.average_cost), 
    0
  );
  
  // Get low stock items
  const { data: lowStockItems, error: lowStockError } = await supabase
    .from('current_stock')
    .select(`
      *,
      product:product_id(id, product_code, product_name, min_stock_level, max_stock_level)
    `);
  
  if (lowStockError) throw lowStockError;
  
  const lowStockCount = (lowStockItems || []).filter(item => 
    item.product && item.current_quantity <= item.product.min_stock_level
  ).length;
  
  // Get active vehicles
  const { data: activeVehicles, error: vehiclesError } = await supabase
    .from('vehicles')
    .select('id')
    .eq('status', 'ACTIVE')
    .eq('active', true);
  
  if (vehiclesError) throw vehiclesError;
  
  return {
    productsCount: new Set(productsWithStock?.map(p => p.product_id)).size || 0,
    totalInventoryValue: totalValue,
    lowStockCount: lowStockCount,
    activeVehiclesCount: activeVehicles?.length || 0,
  };
};
