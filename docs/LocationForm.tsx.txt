LocationForm.tsx

import React, { useState, useEffect } from 'react';
import { Save } from 'lucide-react';
import { toast } from 'react-hot-toast';
import FormField from '../../ui/FormField';
import { useAppContext } from '../../../context/AppContext';
import { Location } from '../../../lib/types';
import { saveLocation } from '../../../lib/api';

interface LocationFormProps {
  onClose: () => void;
  onSave: () => void;
  editingItem?: Location | null;
}

const LocationForm: React.FC<LocationFormProps> = ({ onClose, onSave, editingItem }) => {
  const { products } = useAppContext();
  
  const [formData, setFormData] = useState<Partial<Location>>({
    location_code: '',
    location_name: '',
    location_type: 'STOCKPILE',
    capacity_tonnes: 0,
    assigned_product_id: null,
    description: '',
    active: true
  });
  
  const [errors, setErrors] = useState<Record<string, string>>({});
  const [isSubmitting, setIsSubmitting] = useState(false);

  useEffect(() => {
    if (editingItem) {
      setFormData(editingItem);
    } else {
      setFormData({
        location_code: '',
        location_name: '',
        location_type: 'STOCKPILE',
        capacity_tonnes: 0,
        assigned_product_id: null,
        description: '',
        active: true
      });
    }
  }, [editingItem]);

  const handleChange = (e: React.ChangeEvent<HTMLInputElement | HTMLSelectElement | HTMLTextAreaElement>) => {
    const { name, value, type } = e.target;
    
    if (name === 'assigned_product_id' && value === '') {
      // Handle null value for assigned_product_id
      setFormData({ ...formData, [name]: null });
    } else if (type === 'number') {
      setFormData({ ...formData, [name]: parseFloat(value) || 0 });
    } else {
      setFormData({ ...formData, [name]: value });
    }
    
    // Clear error when field is changed
    if (errors[name]) {
      setErrors({ ...errors, [name]: '' });
    }
  };

  const validateForm = (): boolean => {
    const newErrors: Record<string, string> = {};
    
    // Required fields
    if (!formData.location_code) newErrors.location_code = 'Location code is required';
    if (!formData.location_name) newErrors.location_name = 'Location name is required';
    if (!formData.location_type) newErrors.location_type = 'Location type is required';
    
    // Numeric validation
    if (formData.capacity_tonnes === undefined || formData.capacity_tonnes < 0) {
      newErrors.capacity_tonnes = 'Capacity must be a positive number';
    }
    
    // If location type is STOCKPILE, assigned product is recommended
    if (formData.location_type === 'STOCKPILE' && !formData.assigned_product_id) {
      // This is a warning, not an error
      // newErrors.assigned_product_id = 'Stockpiles should have an assigned product';
    }
    
    setErrors(newErrors);
    return Object.keys(newErrors).length === 0;
  };

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    
    if (!validateForm()) return;
    
    setIsSubmitting(true);
    
    try {
      // Convert assigned_product_id to number if it's a string
      if (formData.assigned_product_id && typeof formData.assigned_product_id === 'string') {
        formData.assigned_product_id = parseInt(formData.assigned_product_id, 10);
      }
      
      await saveLocation(formData);
      toast.success(editingItem ? 'Location updated successfully' : 'Location added successfully');
      onSave();
    } catch (error) {
      console.error('Error saving location:', error);
      toast.error('Failed to save location');
    } finally {
      setIsSubmitting(false);
    }
  };

  return (
    <form onSubmit={handleSubmit} className="space-y-4">
      <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
        <FormField
          name="location_code"
          label="Location Code"
          type="text"
          value={formData.location_code || ''}
          onChange={handleChange}
          required
          placeholder="00-99"
          error={errors.location_code}
        />
        
        <FormField
          name="location_name"
          label="Location Name"
          type="text"
          value={formData.location_name || ''}
          onChange={handleChange}
          required
          error={errors.location_name}
        />
        
        <FormField
          name="location_type"
          label="Location Type"
          type="select"
          value={formData.location_type || ''}
          onChange={handleChange}
          required
          options={['PRODUCTION', 'STOCKPILE', 'LOADING']}
          error={errors.location_type}
        />
        
        <FormField
          name="capacity_tonnes"
          label="Capacity (tonnes)"
          type="number"
          value={formData.capacity_tonnes || ''}
          onChange={handleChange}
          required
          min={0}
          error={errors.capacity_tonnes}
        />
        
        <FormField
          name="assigned_product_id"
          label="Assigned Product"
          type="select"
          value={formData.assigned_product_id?.toString() || ''}
          onChange={handleChange}
          options={[
            { value: '', label: 'None' },
            ...products.map(p => ({ value: p.id, label: `${p.product_code} - ${p.product_name}` }))
          ]}
          error={errors.assigned_product_id}
        />
      </div>
      
      <FormField
        name="description"
        label="Description"
        type="textarea"
        value={formData.description || ''}
        onChange={handleChange}
        error={errors.description}
      />
      
      <div className="flex justify-end space-x-3 mt-6">
        <button
          type="button"
          onClick={onClose}
          className="px-4 py-2 text-gray-700 bg-gray-200 rounded-lg hover:bg-gray-300 transition-colors"
          disabled={isSubmitting}
        >
          Cancel
        </button>
        <button
          type="submit"
          className="px-4 py-2 bg-orange-600 text-white rounded-lg hover:bg-orange-700 transition-colors flex items-center gap-2 disabled:bg-orange-400 disabled:cursor-not-allowed"
          disabled={isSubmitting}
        >
          <Save className="h-4 w-4" />
          {isSubmitting ? 'Saving...' : 'Save Location'}
        </button>
      </div>
    </form>
  );
};

export default LocationForm;
